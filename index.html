  <script>

    const searchQueries = [

      {

        searchText: "Can you give me details of the goods?",

        suggestions: [

          "What documents do I need for customs clearance?",

          "How do I track my shipment in real-time?"

        ]

      },

      {

        searchText: "Can you create shipment for attached Booking confirmation?",

        suggestions: [

          "What's the estimated delivery time for ocean freight?",

          "How do I calculate shipping costs for my cargo?"

        ]

      },

      {

        searchText: "What are the list of shipments currently available?",

        suggestions: [

          "Can you help me file an insurance claim?",

          "What are the customs regulations for importing electronics?"

        ]

      },

      {

        searchText: "How do I schedule a pickup for my freight?",

        suggestions: [

          "What packaging requirements are needed for hazardous materials?",

          "Can you provide a quote for air freight to Europe?"

        ]

      }

    ];


    let currentIndex = 0;

    const typedText = document.getElementById('typedText');

    const suggestion1 = document.getElementById('suggestion1');

    const suggestion2 = document.getElementById('suggestion2');

    const suggestionItems = document.querySelectorAll('.suggestion-item');


    function typeText(text, callback) {

      let charIndex = 0;

      typedText.textContent = '';

      

      // Fade out suggestions while typing

      suggestionItems.forEach(item => item.classList.add('faded'));

      

      const typingInterval = setInterval(() => {

        if (charIndex < text.length) {

          typedText.textContent += text[charIndex];

          charIndex++;

        } else {

          clearInterval(typingInterval);

          

          // Fade in suggestions after typing completes

          suggestionItems.forEach(item => item.classList.remove('faded'));

          

          if (callback) {

            setTimeout(callback, 2000); // Wait 2 seconds before next query

          }

        }

      }, 50); // Fast typing speed (50ms per character)

    }


    function startAnimation() {

      const currentQuery = searchQueries[currentIndex];

      

      // Update suggestions

      suggestion1.textContent = currentQuery.suggestions[0];

      suggestion2.textContent = currentQuery.suggestions[1];

      

      // Type the search text

      typeText(currentQuery.searchText, () => {

        currentIndex = (currentIndex + 1) % searchQueries.length;

        startAnimation();

      });

    }


    // Start the animation

    startAnimation();

  </script>
